DROP TABLE IF EXISTS opportunity;CREATE TABLE opportunity(cp__name varchar(128), a__title varchar(128), f__price NUMERIC(15,2), f__totalprice NUMERIC(15,2));DROP TABLE IF EXISTS person;CREATE TABLE person(a__person varchar(128), l__person__personid varchar(128), l__person__firstname varchar(128), l__person__lastname varchar(128), f__age NUMERIC(15,2));
COPY opportunity(cp__name, a__title, f__price, f__totalprice) FROM LOCAL '${DATA_SOURCE_DIR}/opportunity.csv' DELIMITER ',';COPY person(a__person, l__person__personid, l__person__firstname, l__person__lastname, f__age) FROM LOCAL '${DATA_SOURCE_DIR}/person.csv' DELIMITER ',';
